SE_Colour_Data 
/*
 * STRUCT: SE_Colour_Data
 *
 *   This data type is used to translate between colour models, and
 *   specifically is used in the SE_ConvertColourToGivenModel() function.
 */
typedef struct
{
SE_Colour_Model colour_model;
union
{
SE_RGB_Data rgb;
SE_CMY_Data cmy;
SE_HSV_Data hsv;
} u;
} SE_Colour_Data;
