/*
 * STRUCT: SE_Property_Code
 *
 *   A tagged union to hold either an attribute code
 *   or a variable code.
 */
typedef struct
{
SE_Property_Code_Type code_type;
union
{
EDCS_Attribute_Code attribute;
SE_Variable_Code variable;
} code;
} SE_Property_Code;
