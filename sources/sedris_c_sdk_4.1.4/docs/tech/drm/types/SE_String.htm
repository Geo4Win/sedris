SE_String 
/*
 * STRUCT: SE_String
 *
 *   This data type specifies a string of SE_Characters along with the
 *   locale needed to properly interpret the characters in the string.
 */
typedef struct
{
SE_Locale locale;
SE_Integer_Unsigned length;
SE_Character *characters;
} SE_String;
