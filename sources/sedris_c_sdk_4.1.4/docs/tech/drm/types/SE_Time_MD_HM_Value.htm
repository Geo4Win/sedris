SE_Time_MD_HM_Value 
/*
 * STRUCT: SE_Time_MD_HM_Value
 *
 *   This data type encapsulates the elements of a single
 *   DATE_MD_AND_TIME_HM time value.
 */
typedef struct
{
SE_Month month;
SE_Short_Integer_Positive day;
SE_Byte_Unsigned hour;
SE_Byte_Unsigned minutes;
} SE_Time_MD_HM_Value;
