SRM_CD_3D_Coordinate 
/*!
 * STRUCT: SRM_CD_3D_Coordinate

 *   SRFT CELESTIODETIC.
 */
typedef struct
{
SRM_Long_Float longitude;

SRM_Long_Float ellipsoidal_height;
} SRM_CD_3D_Coordinate;
