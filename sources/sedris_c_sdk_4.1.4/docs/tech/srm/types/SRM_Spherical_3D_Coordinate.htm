SRM_Spherical_3D_Coordinate 
/*!
 * STRUCT: SRM_Spherical_3D_Coordinate
 *
 *   This non-object data type specifies the 3D coordinate-components for
 *   spherical SRFTs.
 */
typedef struct
{
SRM_Long_Float longitude;
SRM_Long_Float latitude;
SRM_Long_Float radius;
} SRM_Spherical_3D_Coordinate;
