EDCS_String 
/*
 * STRUCT: EDCS_String
 *
 *   This data type specifies a string as an array of EDCS_Characters,
 *   along with the locale needed to properly interpret the characters
 *   in the string.
 */
typedef struct
{
EDCS_Locale locale;
EDCS_Count length;
EDCS_Character *characters;
} EDCS_String;
